-v命令实现

docker run -it -v /宿主机的绝对路径目录:/容器内目录 镜像名 #如果不能写，需要添加–privileged=true

docker run -it -v /宿主机的绝对路径目录:/容器内目录:ro 镜像名 #容器内文件只读
